self join:
Find products in a same category
SELECT 
  p1.name AS product,
  p2.name AS recommended_product
FROM 
  products p1
JOIN 
  products p2
ON 
  p1.category_id = p2.category_id
AND 
  p1.product_id != p2.product_id;
